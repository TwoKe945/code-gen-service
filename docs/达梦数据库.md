

[达梦数据库下载地址](https://eco.dameng.com/download/)

```shell
docker load -i dm8_20241022_x86_rh6_64_single.tar
```

启动命令
```shell
docker run -d -p 5236:5236 --name dmdb \
--restart=always --privileged=true \
-e LD_LIBRARY_PATH=/opt/dmdbms/bin \
-e PAGE_SIZE=32 \
-e EXTENT_SIZE=32 \
-e LOG_SIZE=2048 \
-e UNICODE_FLAG=1 \
-e INSTANCE_NAME=DMSERVER \
-v /home/dmdbms:/opt/dmdbms/data \
dm8_single:dm8_20241022_rev244896_x86_rh6_64
```


```oracle
CREATE TABLE TEST_DB.APP_FEEDBACK (
	FEEDBACK_ID numeric NULL AUTO_INCREMENT,
	USER_ID varchar(15) NULL,
	NICK_NAME varchar(30) NULL,
	CONTENT varchar(500) NULL,
	CREATE_TIME date NULL,
	P_FEEDBACK_ID numeric NULL,
	STATUS numeric NULL,
	SEND_TYPE numeric NULL,
	CLIENT_LAST_SEND_TIME date NULL,
	CONSTRAINT APP_FEEDBACK_PK PRIMARY KEY (FEEDBACK_ID)
);
COMMENT ON TABLE TEST_DB.APP_FEEDBACK IS '问题反馈';
COMMENT ON COLUMN TEST_DB.APP_FEEDBACK.FEEDBACK_ID IS '自增ID';
COMMENT ON COLUMN TEST_DB.APP_FEEDBACK.USER_ID IS '用户ID';
COMMENT ON COLUMN TEST_DB.APP_FEEDBACK.NICK_NAME IS '昵称';
COMMENT ON COLUMN TEST_DB.APP_FEEDBACK.CONTENT IS '反馈内容';
COMMENT ON COLUMN TEST_DB.APP_FEEDBACK.CREATE_TIME IS '创建时间';
COMMENT ON COLUMN TEST_DB.APP_FEEDBACK.P_FEEDBACK_ID IS '父级ID';
COMMENT ON COLUMN TEST_DB.APP_FEEDBACK.STATUS IS '状态 0：未回复 1：已回复';
COMMENT ON COLUMN TEST_DB.APP_FEEDBACK.SEND_TYPE IS '发送类型 0：访客 1：管理员';
COMMENT ON COLUMN TEST_DB.APP_FEEDBACK.CLIENT_LAST_SEND_TIME IS '访客最后发送时间';
```
